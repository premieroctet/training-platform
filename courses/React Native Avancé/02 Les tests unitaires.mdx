#### Jest : exemple

```js
import { arrayToMap } from "./transformers";

describe("arrayToMap", () => {
  it("should returns an object with key ids", () => {
    const items = [
      { id: 1, name: "Tom" },
      { id: 2, name: "Jim" },
    ];

    expect(arrayToMap(items)).toStrictEqual({
      1: { id: 1, name: "Tom" },
      2: { id: 2, name: "Jim" },
    });
  });
});
```
